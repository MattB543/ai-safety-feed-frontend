CREATE TABLE public."content" (
	id serial4 NOT NULL,
	"uuid" uuid DEFAULT gen_random_uuid() NULL,
	title text NOT NULL,
	source_url text NOT NULL,
	source_type text NOT NULL,
	authors _text NOT NULL,
	published_date timestamptz NULL,
	content_snippet text NULL,
	full_content text NULL,
	sentence_summary text NULL,
	paragraph_summary text NULL,
	topics _text NULL,
	image_url text NULL,
	created_at timestamptz DEFAULT now() NULL,
	author_credentials text NULL,
	why_valuable text NULL,
	unique_aspects text NULL,
	score int4 NULL,
	full_content_markdown varchar NULL,
	key_implication text NULL,
	title_norm text GENERATED ALWAYS AS (lower(regexp_replace(title, '\s+'::text, ' '::text, 'g'::text))) STORED NULL,
	comment_count int4 NULL,
	cluster_tag text NULL,
	updated_at timestamptz DEFAULT now() NULL
);
